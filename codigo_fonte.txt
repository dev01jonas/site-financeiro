  const [editingClient, setEditingClient] = useState<any>(null);
  const [form, setForm] = useState({ name: '', email: '', phone: '' });

  const { data: clients = [], isLoading } = useQuery({
    queryKey: ['clients', search],
    queryFn: async () => {
      let query = supabase.from('clients').select('*').order('name');
      if (search) {
        query = query.ilike('name', `%${search}%`);
      }
      const { data, error } = await query;
      if (error) throw error;
      return data;
    },
  });

  const saveMutation = useMutation({
    mutationFn: async (client: { name: string; email: string; phone?: string; id?: string }) => {
      if (client.id) {
        const { error } = await supabase.from('clients').update({
          name: client.name,
          email: client.email,
          phone: client.phone || null,
        }).eq('id', client.id);
        if (error) throw error;
      } else {
        const { error } = await supabase.from('clients').insert({
          name: client.name,
          email: client.email,
          phone: client.phone || null,
        });
        if (error) throw error;
      }
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['clients'] });
      setDialogOpen(false);
      setEditingClient(null);
      setForm({ name: '', email: '', phone: '' });
      toast({ title: 'Cliente salvo com sucesso!' });
    },
    onError: () => {
      toast({ title: 'Erro ao salvar cliente', variant: 'destructive' });

  });

    mutationFn: async (id: string) => {
      const { error } = await supabase.from('clients').delete().eq('id', id);
      if (error) throw error;
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['clients'] });
      toast({ title: 'Cliente removido' });
    },
  });

  const importMutation = useMutation({
    mutationFn: async (file: File) => {
      const text = await file.text();
     const lines = text.split('\n').slice(1);

const parsed = lines.map(line => {
  const cols = line.split(',');

  return {
    name: cols[1], // Cliente

    phone: cols[3] === '-' ? '' : cols[3],
  };
}).filter(c => c.name);

      return parsed.length;
    },

      queryClient.invalidateQueries({ queryKey: ['clients'] });

      toast({ title: err.message || 'Erro ao importar CSV', variant: 'destructive' });

  };

  return (

/>

<Button
  variant="outline"

>
  <Upload className="h-4 w-4 mr-2" />
  Importar CSV
</Button>
          <Button size="sm" onClick={handleNew}>

export default function Cobrancas() {
  const { toast } = useToast();
  const queryClient = useQueryClient();
  const [records, setRecords] = useState<BillingRecord[]>([]);
  const [selected, setSelected] = useState<Set<string>>(new Set());
  const [filter, setFilter] = useState<Filter>('all');

  const [dragOver, setDragOver] = useState(false);

    `Prezado(a) {nome},\n\nInformamos que a sua fatura no valor de R$ {valor} com vencimento em {vencimento} encontra-se pendente.\n\nSolicitamos a gentileza de efetuar o pagamento o mais breve possÃ­vel.\n\nAtenciosamente,\nEquipe Financeira - Modaelli`
  );

  const { data: clients = [] } = useQuery({
    queryKey: ['clients-all'],
    queryFn: async () => {
      const { data } = await supabase.from('clients').select('*');
      return data || [];
    },
  });

  const matchClients = useCallback((extracted: ExtractedRecord[]) => {
    // Track already matched client emails to avoid duplicates
    const usedEmails = new Set<string>();

    return extracted.map((rec): BillingRecord => {
      const match = clients.find((c) => {
        const cName = c.name.toLowerCase().trim();
        const rName = rec.name.toLowerCase().trim();
        return cName === rName || cName.includes(rName) || rName.includes(cName);
      });

      const today = new Date();
      const [d, m, y] = rec.dueDate.split('/');
      const dueDate = new Date(+y, +m - 1, +d);
      const isOverdue = dueDate < today;

      // If matched and email already used, still link but mark
      const email = match?.email;
      if (email) usedEmails.add(email);

      return {
        id: crypto.randomUUID(),
        client_name: rec.name,
        due_date: rec.dueDate,
        amount: rec.amount,
        client_id: match?.id,
        client_email: email,
        status: match ? (isOverdue ? 'overdue' : 'matched') : 'unmatched',
      };
    });
  }, [clients]);

  const processFile = async (file: File) => {
    setUploading(true);
    try {
      const extracted = await extractFromPDF(file);
      if (extracted.length === 0) {
        toast({
          title: 'Nenhum registro encontrado',
          description: 'O padrÃ£o de extraÃ§Ã£o pode precisar de ajuste para o formato do seu PDF.',
          variant: 'destructive',
        });
        return;
      }
      const matched = matchClients(extracted);

      // Auto-select all records that have email linked (matched/overdue)
      const autoSelected = new Set<string>();
      matched.forEach(r => {
        if (r.client_email && r.status !== 'sent') {
          autoSelected.add(r.id);
        }
      });

      setRecords(matched);
      setSelected(autoSelected);

      const matchedCount = matched.filter(r => r.client_email).length;
      const unmatchedCount = matched.length - matchedCount;
      toast({
        title: `${extracted.length} registros extraÃ­dos`,
        description: `${matchedCount} vinculados automaticamente${unmatchedCount > 0 ? `, ${unmatchedCount} sem vÃ­nculo` : ''}`,
      });
    } catch {
      toast({ title: 'Erro ao processar PDF', variant: 'destructive' });

import * as pdfjsLib from 'pdfjs-dist';

pdfjsLib.GlobalWorkerOptions.workerSrc = `https://cdnjs.cloudflare.com/ajax/libs/pdf.js/${pdfjsLib.version}/pdf.worker.min.mjs`;

export interface ExtractedRecord {
  name: string;
  dueDate: string;
  amount: number;
}

export async function extractFromPDF(file: File): Promise<ExtractedRecord[]> {
  const arrayBuffer = await file.arrayBuffer();
  const pdf = await pdfjsLib.getDocument({ data: arrayBuffer }).promise;
  
  let fullText = '';
  for (let i = 1; i <= pdf.numPages; i++) {
    const page = await pdf.getPage(i);
    const content = await page.getTextContent();
    const pageText = content.items
      .map((item: any) => item.str)
      .join(' ');
    fullText += pageText + '\n';
  }

  return parseFinancialText(fullText);
}

function parseFinancialText(text: string): ExtractedRecord[] {
  const records: ExtractedRecord[] = [];
  
  // Pattern 1: Name followed by date and currency value
  // Matches: "JoÃ£o da Silva  15/04/2025  R$ 1.500,00"
  const pattern1 = /([A-ZÃ€-Ãš][a-zÃ -Ãº]+(?:\s+(?:de|da|do|dos|das|e)?\s*[A-ZÃ€-Ãša-zÃ -Ãº]+)*)\s+(\d{2}\/\d{2}\/\d{4})\s+R?\$?\s*([\d.,]+)/gi;
  
  let match;
  while ((match = pattern1.exec(text)) !== null) {
    const name = match[1].trim();
    const dueDate = match[2];
    const amountStr = match[3].replace(/\./g, '').replace(',', '.');
    const amount = parseFloat(amountStr);
    
    if (name.length > 3 && !isNaN(amount) && amount > 0) {
      records.push({ name, dueDate, amount });
    }
  }

  // Pattern 2: Try line-by-line if pattern 1 found nothing
  if (records.length === 0) {
    const lines = text.split('\n').filter(l => l.trim());
    for (const line of lines) {
      const dateMatch = line.match(/(\d{2}\/\d{2}\/\d{4})/);
      const amountMatch = line.match(/R?\$?\s*([\d]{1,3}(?:\.?\d{3})*,\d{2})/);
      
      if (dateMatch && amountMatch) {
        const beforeDate = line.substring(0, line.indexOf(dateMatch[0])).trim();
        const name = beforeDate.replace(/^\d+\s*[-.]?\s*/, '').trim();
        const amountStr = amountMatch[1].replace(/\./g, '').replace(',', '.');
        const amount = parseFloat(amountStr);
        
        if (name.length > 3 && !isNaN(amount)) {
          records.push({
            name,
            dueDate: dateMatch[1],
            amount,
          });
        }
      }
    }
  }

  return records;
}

export function exportToCSV(records: BillingRecord[]): string {
  const header = 'Nome,E-mail,Data Vencimento,Valor,Status\n';
  const rows = records.map(r => 
    `"${r.client_name}","${r.client_email || ''}","${r.due_date}","${r.amount.toFixed(2).replace('.', ',')}","${r.status}"`
  ).join('\n');
  return header + rows;
}

export function downloadCSV(records: BillingRecord[], filename = 'cobrancas.csv') {
  const csv = exportToCSV(records);
  const bom = '\uFEFF';
  const blob = new Blob([bom + csv], { type: 'text/csv;charset=utf-8;' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}

export function copyToClipboard(records: BillingRecord[]) {
  const rows = records.map(r => 
    `${r.client_name}\t${r.client_email || ''}\t${r.due_date}\t${r.amount.toFixed(2).replace('.', ',')}\t${r.status}`
  );
  const header = 'Nome\tE-mail\tData Vencimento\tValor\tStatus';
  const text = header + '\n' + rows.join('\n');
  navigator.clipboard.writeText(text);
}

export function parseClientCSV(csvText: string): Array<{ name: string; email: string; phone?: string }> {
  const lines = csvText.split('\n').filter(l => l.trim());
  if (lines.length < 2) return [];
  
  const headerLine = lines[0].toLowerCase();
  const separator = headerLine.includes(';') ? ';' : ',';
  const headers = headerLine.split(separator).map(h => h.trim().replace(/"/g, ''));
  
  const nameIdx = headers.findIndex(h => h.includes('nome') || h === 'name');
  const emailIdx = headers.findIndex(h => h.includes('email') || h.includes('e-mail'));
  const phoneIdx = headers.findIndex(h => h.includes('telefone') || h.includes('phone') || h.includes('tel'));
  
  if (nameIdx === -1 || emailIdx === -1) return [];
  
  return lines.slice(1).map(line => {
    const cols = line.split(separator).map(c => c.trim().replace(/"/g, ''));
    return {
      name: cols[nameIdx] || '',
      email: cols[emailIdx] || '',
      phone: phoneIdx >= 0 ? cols[phoneIdx] : undefined,
    };
  }).filter(c => c.name && c.email);
}
